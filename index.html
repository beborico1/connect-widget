const AWS = require('aws-sdk');

const transcribeservice = new AWS.TranscribeService();

exports.handler = async (event) => {

    console.log("Evento recibido:", JSON.stringify(event, null, 2)); // Muestra el evento entrante

    if (!event.Details || !event.Details.ContactData || !event.Details.ContactData.MediaStreams || !event.Details.ContactData.MediaStreams.Customer || !event.Details.ContactData.MediaStreams.Customer.Audio) {

        console.error("Error: Propiedades necesarias faltantes en el evento");

        return {

            statusCode: 400,

            body: JSON.stringify('Propiedades necesarias faltantes en el evento'),

        };

    }

    const streamARN = event.Details.ContactData.MediaStreams.Customer.Audio.StreamARN;

    const startFragmentNumber = event.Details.ContactData.MediaStreams.Customer.Audio.StartFragmentNumber;

    const contactId = event.Details.ContactData.ContactId;

    if (!streamARN || !startFragmentNumber || !contactId) {

        console.error("Error: Falta ARN del stream, número de fragmento inicial o ContactId");

        return {

            statusCode: 400,

            body: JSON.stringify('Falta información crucial para la transcripción'),

        };

    }

    console.log("ARN del Stream:", streamARN);

    console.log("Número de Fragmento Inicial:", startFragmentNumber);

    console.log("ContactId:", contactId);

    try {

        const transcribeParams = {

            LanguageCode: 'en-US',

            MediaSampleRateHertz: 8000,

            MediaFormat: 'audio/x-l16',

            StreamARN: streamARN,

            StartFragmentNumber: startFragmentNumber

        };

        console.log("Parámetros de Transcripción:", JSON.stringify(transcribeParams, null, 2));

        const transcriptionStream = transcribeservice.startStreamTranscription(transcribeParams);

        transcriptionStream.on('data', (transcript) => {

            try {

                console.log('Datos de transcripción recibidos:', JSON.stringify(transcript, null, 2));

                sendToWebSocket(transcript, contactId);

            } catch (err) {

                console.error('Error al enviar la transcripción al WebSocket:', err);

            }

        });

        transcriptionStream.on('end', () => {

            console.log('Flujo de transcripción finalizado');

        });

        transcriptionStream.on('error', (err) => {

            console.error('Error en el flujo de transcripción:', err);

        });

    } catch (err) {

        console.error('Error al configurar la transcripción:', err);

        return {

            statusCode: 500,

            body: JSON.stringify('Error al configurar el servicio de transcripción'),

        };

    }

    return {

        statusCode: 200,

        body: JSON.stringify('Transcripción iniciada'),

    };

};

async function sendToWebSocket(transcript, contactId) {

    const apiGateway = new AWS.ApiGatewayManagementApi({

        endpoint: 'wss://url-de-websocket.com/@connections/'

    });

    try {

        const postData = { message: transcript, connectionId: contactId }; // Utilizar contactId como connectionId

        console.log("Enviando al WebSocket:", JSON.stringify(postData, null, 2));

        await apiGateway.postToConnection(postData).promise();

        console.log("Publicación en WebSocket exitosa");

    } catch (err) {

        console.error('Error al publicar en WebSocket:', err);

        // Aquí se podría agregar lógica de reintento o alertas

    }

}

# Inicializa el cliente de Lambda

def lambda_handler(event, context):

    try:

        print("Evento recibido de Amazon Connect: " + json.dumps(event))

    except Exception as e:

        print(f"Error al imprimir el evento: {str(e)}")

   

    try:

        # Verifica si es un evento de tipo específico o no

        if event.get('Name') == 'ContactFlowEvent':

            print("Extrayendo carga útil de los detalles del evento.")

            details = event['Details']

            contact_data = details['ContactData']

            attributes = contact_data['Attributes']

            media_streams = contact_data['MediaStreams']['Customer']['Audio']

           

            payload = {

                "streamARN": media_streams['StreamARN'],

                "startFragmentNum": media_streams['StartFragmentNumber'],

                "connectContactId": contact_data['ContactId'],

                "transcriptionEnabled": True if attributes['transcribeCall'] == 'true' else False,

                "saveCallRecording": True if attributes['saveCallRecording'] == 'true' else False,

                "languageCode": attributes['languageCode'],

                "streamAudioFromCustomer": True if attributes['streamAudioFromCustomer'] == 'true' else False,

                "streamAudioToCustomer": True if attributes['streamAudioToCustomer'] == 'true' else False

            }

            print("Carga útil extraída:", json.dumps(payload))

        else:

            print("Tipo de evento no soportado recibido.")

            return {'statusCode': 400, 'body': json.dumps({'error': "Tipo de evento no soportado"})}

    except KeyError as e:

        print(f"Error de clave al extraer datos del evento: {str(e)}")

        return {'statusCode': 400, 'body': json.dumps({'error': "Datos del evento malformados"})}

    print("Parámetros preparados para la invocación de Lambda.")

    params = {

        'FunctionName': os.environ.get('transcriptionFunction', 'defaultFunctionName'),

        'InvocationType': 'Event',  # Invocación asincrónica

        'Payload': json.dumps(payload)

    }

    try:

        print("Invocando otra función Lambda.")

        response = lambda_client.invoke(**params)

        print("Respuesta de la invocación de Lambda: ", json.dumps(response))

        return build_response()

    except Exception as e:

        print(f"Error al invocar la función Lambda: {str(e)}")

        return {'statusCode': 500, 'body': json.dumps({'error': "Fallo al invocar la función"})}

def build_response():

    try:

        response = {

            'statusCode': 200,

            'body': json.dumps({'resultadoLambda': "Éxito"})

        }

        print("Respuesta construida con éxito: ", json.dumps(response))

        return response

    except Exception as e:

        print(f"Error al construir la respuesta: {str(e)}")

        return {'statusCode': 500, 'body': json.dumps({'error': "Fallo al construir la respuesta"})}

import json

import boto3

import os

print('Cargando función...')

# Inicializa el cliente de Transcribe y ApiGatewayManagementApi

transcribe_client = boto3.client('transcribe')

websocket_url = os.environ.get('WEBSOCKET_URL')

connection_id = os.environ.get('CONNECTION_ID')

if not connection_id:

    print("Faltan variables de entorno necesarias: CONNECTION_ID.")

    raise EnvironmentError("Faltan variables de entorno necesarias (CONNECTION_ID).")

if not websocket_url:

    print("Faltan variables de entorno necesarias: WEBSOCKET_URL.")

    raise EnvironmentError("Faltan variables de entorno necesarias (WEBSOCKET_URL).")

api_gateway_client = boto3.client('apigatewaymanagementapi', endpoint_url=websocket_url)

def lambda_handler(event, context):

    print('Evento recibido:', json.dumps(event, indent=2))

    # Asumiendo que el payload ya ha sido extraído como parte del evento

    try:

        stream_arn = event['streamARN']

        start_fragment_number = event['startFragmentNum']

        language_code = event['languageCode']

        transcription_job_name = f"transcripcion-{event['connectContactId']}"

        transcribe_stream = transcribe_client.start_stream_transcription(

            LanguageCode=language_code,

            MediaSampleRateHertz=8000,  # Asegúrate de que esta tasa de muestreo coincide con tu configuración de audio

            MediaFormat='pcm',  # Suponiendo que el formato de entrada es pcm

            AudioStream=stream_arn,

            TranscriptionJobName=transcription_job_name

        )

        # Procesar el stream de transcripción

        for event in transcribe_stream['TranscriptResultStream']:

            print('Transcripción recibida:', json.dumps(event))

            # Enviar la transcripción a WebSocket

            send_to_websocket(event)

        return {

            'statusCode': 200,

            'body': json.dumps('Transcripción iniciada')

        }

    except Exception as e:

        print("Error procesando la transcripción:", e)

        return {

            'statusCode': 500,

            'body': json.dumps('Error al iniciar la transcripción')

        }

def send_to_websocket(data):

    try:

        api_gateway_client.post_to_connection(

            Data=json.dumps(data),

            ConnectionId=connection_id

        )

        print("Datos enviados a WebSocket con éxito.")

    except Exception as e:

        print("Error al enviar datos a través de WebSocket:", e)

        raise

Ejemplo de integración en React:

import React, { useEffect, useState } from 'react';

const AgentsDashboard = () => {

    const [agents, setAgents] = useState([]);

    const [ws, setWs] = useState(null);

    useEffect(() => {

        const websocket = new WebSocket('wss://endpoint-de-api-gateway');

        websocket.onopen = () => {

            console.log('¡WebSocket Conectado!');

        };

        websocket.onmessage = (event) => {

            const message = JSON.parse(event.data);

            // Verificar el tipo de mensaje y actuar en consecuencia

            switch (message.type) {

                case "initialAgentList":

                    // Actualizar el estado con la lista inicial de agentes

                    setAgents(message.data);

                    break;

                case "agentStatusUpdate":

                    // Actualizar un agente específico con un nuevo estado

                    setAgents((currentAgents) => {

                        return currentAgents.map((agent) => {

                            if (agent.agentId === message.data.agentId) {

                                // Actualiza la información del agente con los nuevos datos

                                return { ...agent, ...message.data };

                            }

                            return agent;

                        });

                    });

                    break;

                case "callUpdate":

                    // Actualizar la información de la llamada en curso para un agente específico

                    setAgents((currentAgents) => {

                        return currentAgents.map((agent) => {

                            if (agent.agentId === message.data.agentId) {

                                // Actualiza la duración de la llamada y el análisis de sentimiento

                                return { ...agent, duration: message.data.duration, sentimentScore: message.data.sentimentScore };

                            }

                            return agent;

                        });

                    });

                    break;

                case "callEnded":

                    // Manejar la finalización de la llamada

                    setAgents((currentAgents) => {

                        return currentAgents.map((agent) => {

                            if (agent.agentId === message.data.agentId) {

                                // Restablecer la duración y actualizar el estado y el análisis de sentimiento final

                                return { ...agent, status: "Disponible", duration: message.data.duration, sentimentScore: message.data.sentimentScore };

                            }

                            return agent;

                        });

                    });

                    break;

                default:

                    // Manejar otros tipos de mensajes o casos no contemplados

                    console.log("Tipo de mensaje no reconocido:", message.type);

            }

        };

        websocket.onclose = () => {

            console.log('WebSocket Desconectado');

        };

        websocket.onerror = (error) => {

            console.log('Error de WebSocket: ', error);

        };

        setWs(websocket);

        // Limpiar la conexión WebSocket cuando el componente se desmonta

        return () => {

            websocket.close();

        };

    }, []);

    return (

        <div>

            <h1>Agentes de Amazon Connect:</h1>

            <ul>

                {agents.map((agent, index) => (

                    <li key={index}>

                        {agent.name} - {agent.status}

                    </li>

                ))}

            </ul>

        </div>

    );

};

export default AgentsDashboard;
